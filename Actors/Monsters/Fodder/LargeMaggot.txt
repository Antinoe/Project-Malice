ACTOR LargeMaggot
{
	Health 1
	Radius 4
	Height 5
	Mass 100
	Speed 4
	PainChance 64
	MeleeRange 10
	GibHealth 100
	Obituary "%o was nibbled by a Maggot."
	MONSTER
    DamageFactor "Caco", 0
	+NOBLOOD
	+DONTHURTSPECIES
    +NOINFIGHTING
	+NOTARGET
	+THRUSPECIES
	-SOLID
	-COUNTKILL
	States
	{
	Spawn:
		MAGO A 8 A_Wander
		MAGO B 8 A_Wander
		TNT1 A 0 A_Jump(40,"See")
		Loop
	See:
		MAGO A 5 A_Chase
		MAGO B 5 A_Chase
		TNT1 A 0 A_Jump(7,"Death")
		Loop
	Melee:
			TNT1 A 0 A_Jump(60,"Death")
    		MAGO A 20 A_FaceTarget
    		MAGO B 3 A_CustomMeleeAttack(1,0,0,0,0)
    		Goto See
	Pain:
		TNT1 A 0 
		Goto Death
	Death:
		TNT1 A 0 A_PlaySound("RatDead/Death")
		MAGO BBAA 1
		Stop

	}
}
