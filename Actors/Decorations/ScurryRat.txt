
Actor PM_ScurryRat
{
  radius 8
  height 8
  mass 50
  speed 16
  scale 0.5
  health 1
  seesound 	"rat/squeek"
  activesound	"rat/active"
  deathsound	"rat/death"
  +FLOORCLIP
  +FRIGHTENED
  +LOOKALLAROUND
  +NEVERRESPAWN
  +AMBUSH
  +VULNERABLE
  +TOUCHY
  states
  {
  Spawn:
    RATA AAAAAAAAAAAA 10 A_Look 
	RATA ABC 10
	{
	A_Wander;
	A_Look;
	}
    loop
  See:
    RATA A 2 A_Chase
	RATA A 0 A_CheckSight ("Vanish")
	RATA A 2 A_Chase
	RATA A 0 A_CheckSight ("Vanish")
	RATA B 2 A_Chase
	RATA A 0 A_CheckSight ("Vanish")
	RATA B 2 A_Chase
	RATA A 0 A_CheckSight ("Vanish")
	RATA C 2 A_Chase
	RATA A 0 A_CheckSight ("Vanish")
	RATA C 2 A_Chase
	RATA A 0 A_CheckSight ("Vanish")
	TNT1 A 0 A_Jump(10,"Vanish")
    loop
  Vanish:
    TNT1 A 1
    stop
  Death:
    RATD A 1 A_ScreamAndUnblock
	RATD B 1
	RATD C 1
	stop
  }
}
